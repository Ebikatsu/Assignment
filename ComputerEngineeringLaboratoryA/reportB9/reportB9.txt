・C言語版プログラムのリスト
#include<stdio.h>
#include<stdlib.h>

int fact(int n);

int main(void)
{
	int i;
	for(i = 10; i >= 0; i--)printf("%d!=%d\n", i, fact(i));
	return EXIT_SUCCESS;
}

int fact(int n)
{
	//if(n == 0)return 1;
	//else return n * fact(n -1);
	int i, result = 1;
	for(i = 1; i <= n; i++)result *= i;
	return result;
}
・完成したプログラムのリスト
	.data
facteq:	.asciiz	"!="
enter:	.asciiz	"\n"
	.text
	.align	2
	.globl	main
main:			#main関数は課題B8と同じ。
	li	$t0,10
loop:
	move	$a0,$t0
	li	$v0,1
	syscall
	la	$a0,facteq
	li	$v0,4
	syscall
	move	$a0,$t0
	addi	$sp,$sp,-8
	sw	$t0,4($sp)
	sw	$ra,0($sp)
	jal	fact
	lw	$ra,0($sp)
	lw	$t0,4($sp)
	addi	$sp,$sp,8
	move	$a0,$v0
	li	$v0,1
	syscall
	la	$a0,enter
	li	$v0,4
	syscall
	addi	$t0,$t0,-1
	bge	$t0,$zero,loop
	li	$v0,10
	syscall
fact:				#繰り返し処理を使ったサブルーチン
	move $t0,$a0
	li	$v0,1
	li	$t1,1
floop:
	mult	$v0,$t1
	mflo	$v0
	addi	$t1,$t1,1
	ble	$t1,$t0,floop
	j	$ra
・プログラムの説明
mainにあたる部分は課題B8と同様である。サブルーチンfactでは、$v0に1を代入しておき、$t1を1から$a0まで代入して繰り返し処理をして、その繰り返し処理の内部で$v0に$t1をかけていくことで階乗を求めている。
